## 核心表结构 SQL

```sql
-- 集群表
CREATE TABLE clusters (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(64) NOT NULL UNIQUE COMMENT '集群名称',
    display_name VARCHAR(128) COMMENT '显示名称',
    api_server VARCHAR(255) NOT NULL COMMENT 'K8s API Server 地址',
    kubeconfig_encrypted TEXT COMMENT '加密存储的 kubeconfig',
    sa_token_encrypted TEXT COMMENT 'ServiceAccount Token (加密)',
    ca_cert TEXT COMMENT 'CA 证书',
    status TINYINT DEFAULT 1 COMMENT '状态: 0-禁用 1-启用',
    description VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_name (name)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='K8s 集群信息';

-- 命名空间表
CREATE TABLE namespaces (
    id INT PRIMARY KEY AUTO_INCREMENT,
    cluster_id INT NOT NULL COMMENT '所属集群ID',
    name VARCHAR(64) NOT NULL COMMENT 'K8s Namespace 名称',
    display_name VARCHAR(128) COMMENT '显示名称',
    env_type ENUM('dev', 'test', 'staging', 'prod') NOT NULL COMMENT '环境类型',
    status TINYINT DEFAULT 1,
    description VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (cluster_id) REFERENCES clusters(id) ON DELETE CASCADE,
    UNIQUE KEY uk_cluster_ns (cluster_id, name),
    INDEX idx_env_type (env_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='K8s 命名空间';

-- 服务表
CREATE TABLE services (
    id INT PRIMARY KEY AUTO_INCREMENT,
    namespace_id INT NOT NULL COMMENT '所属命名空间ID',
    name VARCHAR(64) NOT NULL COMMENT '服务名称',
    display_name VARCHAR(128),
    type ENUM('deployment', 'statefulset') DEFAULT 'deployment' COMMENT '工作负载类型',
    deploy_name VARCHAR(128) NOT NULL COMMENT 'K8s Deployment/StatefulSet 名称',
    container_name VARCHAR(128) COMMENT '容器名称',
    harbor_project VARCHAR(64) COMMENT 'Harbor 项目名',
    harbor_repo VARCHAR(128) COMMENT 'Harbor 仓库名',
    port INT COMMENT '服务端口',
    replicas INT DEFAULT 1 COMMENT '副本数',
    resource_limits JSON COMMENT '资源限制配置',
    health_check_path VARCHAR(128) DEFAULT '/health' COMMENT '健康检查路径',
    status TINYINT DEFAULT 1,
    description VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (namespace_id) REFERENCES namespaces(id) ON DELETE CASCADE,
    UNIQUE KEY uk_ns_service (namespace_id, name),
    INDEX idx_deploy_name (deploy_name)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='微服务信息';

-- 发布记录表
CREATE TABLE release_records (
    id INT PRIMARY KEY AUTO_INCREMENT,
    service_id INT NOT NULL COMMENT '服务ID',
    operator_id INT NOT NULL COMMENT '操作人ID',
    version VARCHAR(64) NOT NULL COMMENT '版本号',
    image_tag VARCHAR(128) NOT NULL COMMENT '镜像标签',
    image_full_path VARCHAR(255) COMMENT '完整镜像路径',
    previous_image VARCHAR(255) COMMENT '发布前镜像',
    status ENUM('pending', 'approving', 'running', 'success', 'failed', 'rolled_back') DEFAULT 'pending' COMMENT '状态',
    strategy ENUM('rolling', 'recreate', 'canary') DEFAULT 'rolling' COMMENT '发布策略',
    message TEXT COMMENT '发布信息/错误信息',
    pod_status JSON COMMENT 'Pod 状态快照',
    rollback_to INT COMMENT '回滚到的记录ID',
    approved_by INT COMMENT '审批人ID',
    approved_at TIMESTAMP NULL COMMENT '审批时间',
    started_at TIMESTAMP NULL COMMENT '开始时间',
    completed_at TIMESTAMP NULL COMMENT '完成时间',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (service_id) REFERENCES services(id),
    FOREIGN KEY (operator_id) REFERENCES users(id),
    FOREIGN KEY (rollback_to) REFERENCES release_records(id),
    INDEX idx_service_created (service_id, created_at),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='发布记录';

-- 用户表
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(64) NOT NULL UNIQUE COMMENT '用户名',
    password_hash VARCHAR(255) NOT NULL COMMENT '密码哈希',
    email VARCHAR(128) NOT NULL UNIQUE,
    real_name VARCHAR(64) COMMENT '真实姓名',
    role ENUM('admin', 'developer', 'viewer', 'approver') DEFAULT 'developer' COMMENT '全局角色',
    status TINYINT DEFAULT 1 COMMENT '状态: 0-禁用 1-启用',
    last_login_at TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_role (role)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 权限表 (RBAC)
CREATE TABLE permissions (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL COMMENT '用户ID',
    cluster_id INT COMMENT '集群ID (NULL表示所有集群)',
    namespace_id INT COMMENT '命名空间ID (NULL表示所有命名空间)',
    role ENUM('admin', 'developer', 'viewer') NOT NULL COMMENT '角色权限',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (cluster_id) REFERENCES clusters(id) ON DELETE CASCADE,
    FOREIGN KEY (namespace_id) REFERENCES namespaces(id) ON DELETE CASCADE,
    UNIQUE KEY uk_user_resource (user_id, cluster_id, namespace_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='权限表';

-- 审计日志表
CREATE TABLE audit_logs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id INT COMMENT '操作用户ID',
    username VARCHAR(64) COMMENT '用户名(冗余)',
    action VARCHAR(32) NOT NULL COMMENT '操作类型: CREATE/UPDATE/DELETE/RELEASE/ROLLBACK',
    resource_type VARCHAR(32) NOT NULL COMMENT '资源类型: cluster/namespace/service/release',
    resource_id INT COMMENT '资源ID',
    resource_name VARCHAR(128) COMMENT '资源名称',
    detail JSON COMMENT '操作详情',
    ip_addr VARCHAR(64) COMMENT 'IP地址',
    user_agent VARCHAR(255) COMMENT 'User-Agent',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_created (user_id, created_at),
    INDEX idx_action (action),
    INDEX idx_resource (resource_type, resource_id),
    INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='审计日志';
```

```python
# app/core/k8s_client.py
import asyncio
import logging
from typing import Optional, Dict, Any, Callable
from contextlib import asynccontextmanager

from kubernetes import client, config, watch
from kubernetes.client.exceptions import ApiException
from kubernetes.config.config_exception import ConfigException

logger = logging.getLogger(__name__)


class K8sClientManager:
    """K8s 客户端管理器 - 支持多集群"""
    
    _instances: Dict[int, 'K8sClientManager'] = {}
    
    def __init__(self, cluster_id: int, api_server: str, token: str, ca_cert: Optional[str] = None):
        self.cluster_id = cluster_id
        self.api_server = api_server
        self.token = token
        self.ca_cert = ca_cert
        self._apps_v1: Optional[client.AppsV1Api] = None
        self._core_v1: Optional[client.CoreV1Api] = None
        self._lock = asyncio.Lock()
    
    @classmethod
    async def get_client(cls, cluster_id: int, api_server: str, token: str, ca_cert: Optional[str] = None) -> 'K8sClientManager':
        """获取或创建客户端实例 (单例模式)"""
        if cluster_id not in cls._instances:
            cls._instances[cluster_id] = cls(cluster_id, api_server, token, ca_cert)
            await cls._instances[cluster_id]._init_client()
        return cls._instances[cluster_id]
    
    async def _init_client(self):
        """初始化 K8s 客户端"""
        try:
            configuration = client.Configuration()
            configuration.host = self.api_server
            configuration.verify_ssl = True
            configuration.api_key['authorization'] = f'Bearer {self.token}'
            
            if self.ca_cert:
                configuration.ssl_ca_cert = self._write_temp_cert(self.ca_cert)
            
            client.Configuration.set_default(configuration)
            self._apps_v1 = client.AppsV1Api()
            self._core_v1 = client.CoreV1Api()
            logger.info(f"K8s client initialized for cluster {self.cluster_id}")
        except Exception as e:
            logger.error(f"Failed to initialize K8s client: {e}")
            raise K8sClientError(f"K8s client init failed: {e}")
    
    def _write_temp_cert(self, cert_content: str) -> str:
        """将证书写入临时文件"""
        import tempfile
        with tempfile.NamedTemporaryFile(mode='w', suffix='.crt', delete=False) as f:
            f.write(cert_content)
            return f.name
    
    @property
    def apps_v1(self) -> client.AppsV1Api:
        if not self._apps_v1:
            raise K8sClientError("K8s client not initialized")
        return self._apps_v1
    
    @property
    def core_v1(self) -> client.CoreV1Api:
        if not self._core_v1:
            raise K8sClientError("K8s client not initialized")
        return self._core_v1


class K8sClientError(Exception):
    """K8s 客户端异常"""
    pass


class DeploymentUpdateError(Exception):
    """Deployment 更新异常"""
    pass

        self,
        namespace: str,
        deployment_name: str,
        container_name: str,
        new_image: str,
        timeout: int = 300,
        progress_callback: Optional[Callable[[Dict[str, Any]], None]] = None
    ) -> Dict[str, Any]:
        """
        更新 Deployment 镜像并等待滚动更新完成
        
        Args:
            namespace: 命名空间
            deployment_name: Deployment 名称
            container_name: 容器名称
            new_image: 新镜像地址
            timeout: 超时时间(秒)
            progress_callback: 进度回调函数
            
        Returns:
            更新结果信息
        """
        client_mgr = await self._get_client()
        apps_v1 = client_mgr.apps_v1
        
        try:
            # 1. 获取当前 Deployment
            logger.info(f"Fetching deployment {deployment_name} in namespace {namespace}")
            deployment = await asyncio.to_thread(
                apps_v1.read_namespaced_deployment,
                name=deployment_name,
                namespace=namespace
            )
            
            # 2. 记录原始镜像信息
            original_image = None
            container_found = False
            for container in deployment.spec.template.spec.containers:
                if container.name == container_name:
                    original_image = container.image
                    container.image = new_image
                    container_found = True
                    break
            
            if not container_found:
                raise DeploymentUpdateError(f"Container {container_name} not found in deployment")
            
            # 3. 添加版本注解 (用于追踪)
            if deployment.spec.template.metadata.annotations is None:
                deployment.spec.template.metadata.annotations = {}
            deployment.spec.template.metadata.annotations['release-platform/version'] = new_image
            deployment.spec.template.metadata.annotations['release-platform/updated-at'] = datetime.utcnow().isoformat()
            
            # 4. 执行更新
            logger.info(f"Updating deployment image: {original_image} -> {new_image}")
            updated_deployment = await asyncio.to_thread(
                apps_v1.patch_namespaced_deployment,
                name=deployment_name,
                namespace=namespace,
                body=deployment
            )
            
            # 5. 等待滚动更新完成
            result = await self._wait_for_rolling_update(
                namespace=namespace,
                deployment_name=deployment_name,
                timeout=timeout,
                progress_callback=progress_callback
            )
            
            return {
                'success': result['success'],
                'deployment': updated_deployment.metadata.name,
                'namespace': namespace,
                'original_image': original_image,
                'new_image': new_image,
                'replicas': updated_deployment.spec.replicas,
                'ready_replicas': result.get('ready_replicas', 0),
                'message': result.get('message', ''),
                'duration': result.get('duration', 0)
            }
            
        except ApiException as e:
            logger.error(f"K8s API error: {e.status} - {e.reason}")
            if e.status == 404:
                raise DeploymentUpdateError(f"Deployment {deployment_name} not found in namespace {namespace}")
            elif e.status == 403:
                raise DeploymentUpdateError(f"Permission denied: {e.reason}")
            else:
                raise DeploymentUpdateError(f"K8s API error: {e.reason}")
        except asyncio.TimeoutError:
            raise DeploymentUpdateError(f"Rolling update timeout after {timeout}s")
        except Exception as e:
            logger.error(f"Unexpected error during deployment update: {e}")
            raise DeploymentUpdateError(f"Update failed: {str(e)}")
    
    async def _wait_for_rolling_update(
        self,
        namespace: str,
        deployment_name: str,
        timeout: int,
        progress_callback: Optional[Callable[[Dict[str, Any]], None]] = None
    ) -> Dict[str, Any]:
        """
        等待滚动更新完成
        
        监控指标:
        - UpdatedReplicas: 已更新副本数
        - ReadyReplicas: 就绪副本数
        - AvailableReplicas: 可用副本数
        - UnavailableReplicas: 不可用副本数
        """
        client_mgr = await self._get_client()
        apps_v1 = client_mgr.apps_v1
        
        start_time = asyncio.get_event_loop().time()
        check_interval = 3  # 检查间隔(秒)
        
        while True:
            elapsed = asyncio.get_event_loop().time() - start_time
            if elapsed > timeout:
                raise asyncio.TimeoutError()
            
            try:
                # 获取 Deployment 状态
                deployment = await asyncio.to_thread(
                    apps_v1.read_namespaced_deployment_status,
                    name=deployment_name,
                    namespace=namespace
                )
                
                status = deployment.status
                spec = deployment.spec
                
                # 计算进度
                desired = spec.replicas or 0
                updated = status.updated_replicas or 0
                ready = status.ready_replicas or 0
                available = status.available_replicas or 0
                unavailable = status.unavailable_replicas or 0
                
                progress = {
                    'desired': desired,
                    'updated': updated,
                    'ready': ready,
                    'available': available,
                    'unavailable': unavailable,
                    'progress_percent': (ready / desired * 100) if desired > 0 else 0,
                    'elapsed_seconds': int(elapsed),
                    'status': 'updating'
                }
                
                # 检查更新条件
                conditions = status.conditions or []
                progressing = next(
                    (c for c in conditions if c.type == 'Progressing'),
                    None
                )
                
                if progressing and progressing.reason == 'ProgressDeadlineExceeded':
                    progress['status'] = 'failed'
                    progress['message'] = 'Progress deadline exceeded'
                    if progress_callback:
                        await asyncio.to_thread(progress_callback, progress)
                    return {
                        'success': False,
                        'message': 'Rolling update failed: progress deadline exceeded',
                        'ready_replicas': ready,
                        'duration': int(elapsed)
                    }
                
                # 检查是否完成
                if updated == desired and ready == desired and available == desired:
                    progress['status'] = 'completed'
                    if progress_callback:
                        await asyncio.to_thread(progress_callback, progress)
                    return {
                        'success': True,
                        'message': 'Rolling update completed successfully',
                        'ready_replicas': ready,
                        'duration': int(elapsed)
                    }
                
                # 回调进度
                if progress_callback:
                    await asyncio.to_thread(progress_callback, progress)
                
                logger.debug(f"Rolling update progress: {ready}/{desired} ready")
                
            except ApiException as e:
                logger.error(f"Error checking deployment status: {e}")
                raise
            
            await asyncio.sleep(check_interval)
    
    async def rollback_deployment(
        self,
        namespace: str,
        deployment_name: str,
        target_revision: Optional[int] = None,
        timeout: int = 300
    ) -> Dict[str, Any]:
        """
        回滚 Deployment 到指定版本
        
        Args:
            namespace: 命名空间
            deployment_name: Deployment 名称
            target_revision: 目标版本号 (None 表示回滚到上一个版本)
            timeout: 超时时间
        """
        client_mgr = await self._get_client()
        apps_v1 = client_mgr.apps_v1
        
        try:
            # 获取当前 Deployment
            deployment = await asyncio.to_thread(
                apps_v1.read_namespaced_deployment,
                name=deployment_name,
                namespace=namespace
            )
            
            # 获取当前镜像
            current_image = None
            if deployment.spec.template.spec.containers:
                current_image = deployment.spec.template.spec.containers[0].image
            
            # 执行回滚 (通过设置 annotation 触发)
            if deployment.metadata.annotations is None:
                deployment.metadata.annotations = {}
            
            deployment.metadata.annotations['deployment.kubernetes.io/revision'] = str(target_revision or '')
            
            # 注意: Python client 没有直接的 rollback 方法
            # 实际项目中应该:
            # 1. 从 release_records 表获取历史镜像
            # 2. 调用 update_deployment_image 恢复到历史镜像
            
            logger.info(f"Rollback initiated for {deployment_name}")
            
            return {
                'success': True,
                'message': 'Rollback initiated',
                'deployment': deployment_name,
                'namespace': namespace
            }
            
        except ApiException as e:
            logger.error(f"Rollback failed: {e}")
            raise DeploymentUpdateError(f"Rollback failed: {e.reason}")
    
    async def get_deployment_status(
        self,
        namespace: str,
        deployment_name: str
    ) -> Dict[str, Any]:
        """获取 Deployment 状态"""
        client_mgr = await self._get_client()
        apps_v1 = client_mgr.apps_v1
        core_v1 = client_mgr.core_v1
        
        try:
            # 获取 Deployment
            deployment = await asyncio.to_thread(
                apps_v1.read_namespaced_deployment,
                name=deployment_name,
                namespace=namespace
            )
            
            # 获取 Pod 列表
            selector = deployment.spec.selector.match_labels
            label_selector = ','.join([f"{k}={v}" for k, v in selector.items()])
            
            pods = await asyncio.to_thread(
                core_v1.list_namespaced_pod,
                namespace=namespace,
                label_selector=label_selector
            )
            
            pod_list = []
            for pod in pods.items:
                pod_list.append({
                    'name': pod.metadata.name,
                    'status': pod.status.phase,
                    'ready': f"{sum(1 for c in pod.status.container_statuses or [] if c.ready)}/{len(pod.status.container_statuses or [])}",
                    'restarts': sum(c.restart_count for c in pod.status.container_statuses or []),
                    'age': pod.metadata.creation_timestamp.isoformat() if pod.metadata.creation_timestamp else None,
                    'node': pod.spec.node_name
                })
            
            return {
                'name': deployment.metadata.name,
                'namespace': namespace,
                'replicas': deployment.spec.replicas,
                'ready_replicas': deployment.status.ready_replicas or 0,
                'updated_replicas': deployment.status.updated_replicas or 0,
                'available_replicas': deployment.status.available_replicas or 0,
                'strategy': deployment.spec.strategy.type if deployment.spec.strategy else 'RollingUpdate',
                'pods': pod_list,
                'conditions': [
                    {
                        'type': c.type,
                        'status': c.status,
                        'reason': c.reason,
                        'message': c.message
                    }
                    for c in (deployment.status.conditions or [])
                ]
            }
            
        except ApiException as e:
            logger.error(f"Failed to get deployment status: {e}")
            raise K8sClientError(f"Failed to get status: {e.reason}")
```

```python
# app/services/release_service.py
import asyncio
import json
from datetime import datetime
from typing import Optional, Dict, Any, List

class ReleaseService:
    """发布服务 - 处理完整的发布流程"""
    
    def __init__(self, db: Session):
        self.db = db
        self._active_releases: Dict[int, asyncio.Task] = {}
    
    async def create_release(
        self,
        service_id: int,
        operator_id: int,
        image_tag: str,
        version: Optional[str] = None,
        require_approval: bool = False
    ) -> ReleaseRecord:
        """创建发布单"""
        # 获取服务信息
        service = self.db.query(Service).filter(Service.id == service_id).first()
        if not service:
            raise ValueError(f"Service {service_id} not found")
        
        # 获取当前运行的镜像
        current_image = await self._get_current_image(service)
        
        # 构建完整镜像路径
        image_full_path = self._build_image_path(service, image_tag)
        
        # 创建发布记录
        release = ReleaseRecord(
            service_id=service_id,
            operator_id=operator_id,
            version=version or image_tag,
            image_tag=image_tag,
            image_full_path=image_full_path,
            previous_image=current_image,
            status=ReleaseStatus.PENDING if not require_approval else ReleaseStatus.APPROVING,
            message='Waiting for execution'
        )
        
        self.db.add(release)
        self.db.commit()
        self.db.refresh(release)
        
        logger.info(f"Release created: {release.id} for service {service.name}")
        
        return release
    
    async def execute_release(
        self,
        release_id: int,
        operator_id: int
    ) -> Dict[str, Any]:
        """执行发布"""
        release = self.db.query(ReleaseRecord).filter(ReleaseRecord.id == release_id).first()
        if not release:
            raise ValueError(f"Release {release_id} not found")
        
        if release.status not in [ReleaseStatus.PENDING, ReleaseStatus.APPROVED]:
            raise ValueError(f"Cannot execute release with status: {release.status}")
        
        service = self.db.query(Service).filter(Service.id == release.service_id).first()
        cluster = self.db.query(Cluster).filter(Cluster.id == service.namespace.cluster_id).first()
        
        # 更新状态为运行中
        release.status = ReleaseStatus.RUNNING
        release.started_at = datetime.utcnow()
        self.db.commit()
        
        try:
            # 创建 K8s 服务
            k8s_service = K8sService(
                cluster_id=cluster.id,
                api_server=cluster.api_server,
                token=cluster.sa_token_encrypted,  # 实际应该解密
                ca_cert=cluster.ca_cert
            )
            
            # 定义进度回调
            async def progress_callback(progress: Dict[str, Any]):
                release.pod_status = json.dumps(progress)
                self.db.commit()
                # TODO: 通过 WebSocket 推送进度
            
            # 执行更新
            result = await k8s_service.update_deployment_image(
                namespace=service.namespace.name,
                deployment_name=service.deploy_name,
                container_name=service.container_name or service.deploy_name,
                new_image=release.image_full_path,
                timeout=600,
                progress_callback=progress_callback
            )
            
            # 更新发布记录
            release.status = ReleaseStatus.SUCCESS if result['success'] else ReleaseStatus.FAILED
            release.message = result['message']
            release.completed_at = datetime.utcnow()
            self.db.commit()
            
            return {
                'success': result['success'],
                'release_id': release.id,
                'message': result['message'],
                'duration': result['duration']
            }
            
        except DeploymentUpdateError as e:
            release.status = ReleaseStatus.FAILED
            release.message = str(e)
            release.completed_at = datetime.utcnow()
            self.db.commit()
            
            logger.error(f"Release {release_id} failed: {e}")
            raise
        except Exception as e:
            release.status = ReleaseStatus.FAILED
            release.message = f"Unexpected error: {str(e)}"
            release.completed_at = datetime.utcnow()
            self.db.commit()
            
            logger.error(f"Release {release_id} failed with unexpected error: {e}")
            raise
    
    async def rollback_release(
        self,
        release_id: int,
        operator_id: int
    ) -> Dict[str, Any]:
        """回滚到指定发布版本"""
        release = self.db.query(ReleaseRecord).filter(ReleaseRecord.id == release_id).first()
        if not release:
            raise ValueError(f"Release {release_id} not found")
        
        if not release.previous_image:
            raise ValueError("No previous version available for rollback")
        
        service = self.db.query(Service).filter(Service.id == release.service_id).first()
        cluster = self.db.query(Cluster).filter(Cluster.id == service.namespace.cluster_id).first()
        
        # 创建回滚记录
        rollback_release = ReleaseRecord(
            service_id=release.service_id,
            operator_id=operator_id,
            version=f"rollback-{release.version}",
            image_tag=release.previous_image.split(':')[-1],
            image_full_path=release.previous_image,
            previous_image=release.image_full_path,
            status=ReleaseStatus.RUNNING,
            rollback_to=release.id,
            started_at=datetime.utcnow()
        )
        self.db.add(rollback_release)
        self.db.commit()
        
        try:
            k8s_service = K8sService(
                cluster_id=cluster.id,
                api_server=cluster.api_server,
                token=cluster.sa_token_encrypted,
                ca_cert=cluster.ca_cert
            )
            
            result = await k8s_service.update_deployment_image(
                namespace=service.namespace.name,
                deployment_name=service.deploy_name,
                container_name=service.container_name or service.deploy_name,
                new_image=release.previous_image,
                timeout=600
            )
            
            rollback_release.status = ReleaseStatus.SUCCESS if result['success'] else ReleaseStatus.FAILED
            rollback_release.message = f"Rollback: {result['message']}"
            rollback_release.completed_at = datetime.utcnow()
            
            # 更新原发布记录状态
            release.status = ReleaseStatus.ROLLED_BACK
            self.db.commit()
            
            return {
                'success': result['success'],
                'rollback_release_id': rollback_release.id,
                'message': result['message']
            }
            
        except Exception as e:
            rollback_release.status = ReleaseStatus.FAILED
            rollback_release.message = f"Rollback failed: {str(e)}"
            rollback_release.completed_at = datetime.utcnow()
            self.db.commit()
            raise
```

```typescript
// frontend/src/hooks/useWebSocket.ts
import { useEffect, useRef, useState, useCallback } from 'react';

export const useWebSocket = (url: string | null) => {
  const [lastMessage, setLastMessage] = useState<MessageEvent | null>(null);
  const [readyState, setReadyState] = useState<number>(WebSocket.CONNECTING);
  const wsRef = useRef<WebSocket | null>(null);
  
  useEffect(() => {
    if (!url) return;
    
    const ws = new WebSocket(url);
    wsRef.current = ws;
    
    ws.onopen = () => {
      setReadyState(WebSocket.OPEN);
    };
    
    ws.onmessage = (event) => {
      setLastMessage(event);
    };
    
    ws.onclose = () => {
      setReadyState(WebSocket.CLOSED);
    };
    
    ws.onerror = (error) => {
      console.error('WebSocket error:', error);
      setReadyState(WebSocket.CLOSED);
    };
    
    return () => {
      ws.close();
    };
  }, [url]);
  
  const sendMessage = useCallback((data: string | object) => {
    if (wsRef.current && wsRef.current.readyState === WebSocket.OPEN) {
      const message = typeof data === 'string' ? data : JSON.stringify(data);
      wsRef.current.send(message);
    }
  }, []);
  
  return { lastMessage, readyState, sendMessage };
};
```

### 5.4 敏感信息加密存储方案

```python
# app/core/security.py
from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import base64
import os


class SecretManager:
    """敏感信息加密管理器"""
    
    def __init__(self, master_key: str):
        """
        初始化加密管理器
        
        Args:
            master_key: 主密钥 (应从环境变量或 Vault 获取)
        """
        self.master_key = master_key
        self._fernet = self._create_fernet(master_key)
    
    def _create_fernet(self, key: str) -> Fernet:
        """创建 Fernet 实例"""
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=os.urandom(16),
            iterations=100000,
        )
        key_bytes = base64.urlsafe_b64encode(kdf.derive(key.encode()))
        return Fernet(key_bytes)
    
    def encrypt(self, plaintext: str) -> str:
        """加密明文"""
        if not plaintext:
            return ''
        return self._fernet.encrypt(plaintext.encode()).decode()
    
    def decrypt(self, ciphertext: str) -> str:
        """解密密文"""
        if not ciphertext:
            return ''
        return self._fernet.decrypt(ciphertext.encode()).decode()
    
    @staticmethod
    def generate_key() -> str:
        """生成新的加密密钥"""
        return Fernet.generate_key().decode()


# 使用 HashiCorp Vault 的替代方案
class VaultSecretManager:
    """基于 Vault 的密钥管理"""
    
    def __init__(self, vault_addr: str, vault_token: str):
        import hvac
        self.client = hvac.Client(url=vault_addr, token=vault_token)
    
    def store_kubeconfig(self, cluster_name: str, kubeconfig: str):
        """存储 kubeconfig 到 Vault"""
        self.client.secrets.kv.v2.create_or_update_secret(
            path=f'k8s-platform/clusters/{cluster_name}/kubeconfig',
            secret={'data': kubeconfig}
        )
    
    def get_kubeconfig(self, cluster_name: str) -> str:
        """从 Vault 获取 kubeconfig"""
        response = self.client.secrets.kv.v2.read_secret_version(
            path=f'k8s-platform/clusters/{cluster_name}/kubeconfig'
        )
        return response['data']['data']['data']
```
